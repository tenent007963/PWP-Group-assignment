# Program initialization
START
	DEFINE FUNCTION init()
	DISPLAY Welcome to APU Online Shopping Mall(AOSM)!
	DEFINE filelist WITH users.txt, products.txt, orders.txt, customer.txt
	LOOP THROUGH ELEMENT IN fileList AS f:
		TRY:
			OPEN f with argument os.O_RDWR
			CLOSE f
		EXCEPTION IF ERROR:
			DISPLAY An error occured. Please try to create [f] manually. 
	CALL mainui()
END
